\def\kn@wnmodules{polyglot-simplepages}
\def\modules@ll{all}
\def\module@startif#1{%
  \ifcsname #1@module@loaded\endcsname\else
  \x@\let\csname #1@module@loaded\endcsname\empty
}
\def\module@endif{\fi}
\def\module@check#1\E{%
  \ifcsname #1@module@loaded\endcsname\else
    \input #1.tex
  \fi}

\def\modules@needed#1{% Expects a comma-separated list of modules that should be loaded. For use in modules.
  \let\do=\module@check
  \cstackdown#1,\E 
}

\ifcsname modulelist\endcsname
  \def\d@#1\E{\module@check #1\E}%
  \ifx\modulelist\modules@ll
    \x@\cstackdown\kn@wnmodules,\E 
  \else
    \x@\cstackdown\modulelist,\E 
  \fi
\else
 \lowercase{\message{load optional modules with (some of) |def|modulelist{\kn@wnmodules} or |def|modulelist{all}}}
\fi
